FizzBuzz

Console logs Fizz if the number is divisible by 3, Buzz if the number is divisible by 5, FizzBuzz if the number is divisible by 15, otherwise it logs the number

                  
const fizzBuzz = function(arr) {
    for (let i = 0; i < arr.length; i++) {
        if(arr[i] % 15 ===0){
            console.log("FizzBuzz")
        }else if(arr[i] % 5 ===0){
            console.log("Buzz")
        }else if(arr[i] % 3 ===0){
            console.log("Fizz")
        }else{
            console.log(arr[i])
        }
    }
};

Max Num

Finds the largest number in an array

                  
var maxNum = function(arr) {
    let largestNum = arr[0]
    for (let i = 0; i < arr.length; i++) {
        if(arr[i]> largestNum){
            largestNum = arr[i]
        }
    }
    console.log(Math.max(...arr),largestNum)
    return largestNum
};

Bubble Sort

Runs through an array checking to see which number is larger. Goes from left to right, moving the smaller numbers to the left

                  
const bubbleSort = (array) => {
    for (let i = 0; i < array.length; i++) {
        for (let j = i+1; j < array.length; j++) {
            if(array[i]>array[j]){
            let temp = array[i];
            array[i] = array[j]
            array[j]=temp
            }
        }
    }return array
};

Find and Replace

String Reverse

Turns a given string into an array of letters, reverses it, then joins the array of letters back into a string

                  
const reverseString = str => […str].reverse().join(‘’);
let a = reverseString(‘Have a nice day!’); // !yad ecin a evaH
